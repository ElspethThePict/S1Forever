event ObjectMain
	object.drawOrder = -1
	if object[-playerCount].state == 27
		object.type = TypeName[Blank Object]
	end if
	if object[-playerCount].state == 28
		object.type = TypeName[Blank Object]
	end if
	object.frame = object.value0
	object.frame >>= 1
	object.value0++
	object.value0 %= 10
end event

	end if
end event
