event ObjectMain
		
	switch object.state
	case 0
		temp0 = object.ypos
		temp0 += 0x400000
		if object[0].ypos < temp0
			stage.newXBoundary1 = object.xpos
			stage.newXBoundary1 >>= 16
			stage.newXBoundary1 -= screen.xcenter
			stage.newXBoundary1 -= screen.xcenter
			stage.newXBoundary2 = object.xpos
			stage.newXBoundary2 >>= 16
			stage.newXBoundary2 += screen.xcenter
			if object[0].xpos > object.xpos
				stage.newXBoundary1 = object.xpos
				stage.newXBoundary1 >>= 16
				stage.newXBoundary1 -= screen.xcenter
				if stage.debugMode == 0
					if stage.listPos == 0
						if stage.minutes == 0
							if stage.seconds < 30
								achievement.trigger = 1
								//CallNativeFunction2(SetAchievement, 1, 100)
							end if
						end if
					end if
					if stage.listPos == 7
						temp0 = 230
						temp0 <<= 16
						if object[0].ypos < temp0
							achievement.trigger = 16
						end if
					end if
				end if
				if stage.timeEnabled == 1
					if options.gameMode == 2
						timeAttack.result = stage.seconds
						timeAttack.result *= 100
						temp0 = stage.minutes
						temp0 *= 0x1770
						timeAttack.result += temp0
						timeAttack.result += stage.milliSeconds
							//stage.listPos--
						switch stage.playerListPos
							case 0
								arrayPos0 = 500
								break
							case 1
								arrayPos0 = 530
								break
							case 2
								arrayPos0 = 560
								break
							case 5
								arrayPos0 = 470
								break						
						end switch
						if SuperTimeAttack == 1
							arrayPos0 = 590
						end if
						arraypos0 += stage.listPos
						if timeAttack.result < saveRAM[arrayPos0]
							saveRAM[arrayPos0] = timeAttack.result
							//object[9].value2 = timeAttack.result
							PlaySfx(SfxName[Event], 0)
						end if
						WriteSaveRAM()
					end if
				end if
				stage.timeEnabled = 0
				object.state = 1
				PlaySfx(SfxName[Sign Post], 0)
				if PlayerObject_SuperState == 1
					PlayerObject_SuperState = 2
				end if
			end if
		end if
		break
	case 1
		music.volume--
		object.value1++
		if object.value1 == 2
			object.value1 = 0
			object.frame++
			if object.frame == 5
				object.frame = 1
				Rand(temp0, 48)
				temp0 <<= 16
				temp0 += object.xpos
				temp0 -= 0x180000
				Rand(temp1, 32)
				temp1 <<= 16
				temp1 += object.ypos
				temp1 -= 0x140000
				CreateTempObject(TypeName[Ring Sparkle], 0, temp0, temp1)
				object.value0--
				if object.value0 == 0
					if object[0].type == TypeName[Blank Object]
						object.state = 4
						if object.propertyValue < 3
							PlayMusic(1)
						else
							if object.outOfBounds == 0
								PlayMusic(1)
							end if
						end if
						object[30].type = TypeName[Act Finish]
						object[30].drawOrder = 6
						object[30].priority = PRIORITY_ACTIVE
						object[0].controlMode = -1
						object[0].interaction = 0
						if stage.player2Enabled == 1
							object[1].interaction = 0
							object[1].controlMode = -1
							object[1].up = 0
							object[1].down = 0
							object[1].left = 0
							object[1].jumpPress = 0
							object[1].jumpHold = 0
						end if
					else
						object[0].controlMode = -1
						//object[0].interaction = 0
						object[0].value15 = 0
						object[0].up = 0
						object[0].down = 0
						object[0].left = 0
						object[0].jumpPress = 0
						object[0].jumpHold = 0
						if stage.player2Enabled == 1
							object[1].controlMode = -1
							object[1].interaction = 0
							object[1].up = 0
							object[1].down = 0
							object[1].left = 0
							object[1].jumpPress = 0
							object[1].jumpHold = 0
						end if
						object.state = 2
					end if
				end if
			end if
		end if
		temp0 = 40
		temp0 += screen.xcenter
		temp0 <<= 16
		temp0 += object.xpos
		if object[0].xpos > temp0
			object[0].xpos = temp0
			object[0].speed = 0
			object[0].xvel = 0
		end if
		if stage.player2Enabled == 1
			if object[1].xpos > temp0
				object[1].xpos = temp0
				object[1].speed = 0
				object[1].xvel = 0
			end if
		end if
		break
	case 2
		if object[0].type == TypeName[Blank Object]
			object.state = 4
			if object.propertyValue < 3
				PlayMusic(1)
			else
				if object.outOfBounds == 0
					PlayMusic(1)
				end if
			end if
			object[30].type = TypeName[Act Finish]
			object[30].drawOrder = 6
			object[30].priority = PRIORITY_ACTIVE
		end if
		if object[0].controlMode == -1
			object[0].right = 1
			if object[0].xvel < 0
				object[0].interaction = 1
			else
				object[0].interaction = 0
			end if
		end if
		if stage.player2Enabled == 1
			object[1].controlMode = object[0].controlMode
			if object[1].controlMode == -1
				if object[1].left == 0
					object[1].right = 1
				end if
			end if
		end if
		temp0 = screen.xcenter
		temp0 <<= 16
		temp0 += object.xpos
		if object[0].xpos > temp0
			if object.propertyValue < 3
				PlayMusic(1)
			else
				if object.outOfBounds == 0
					PlayMusic(1)
				end if
			end if
			signpost.timeout = 0
			object.state = 3
			object[30].type = TypeName[Act Finish]
			object[30].drawOrder = 6
			object[30].priority = PRIORITY_ACTIVE
		else
			signpost.timeout++
			if signpost.timeout >= 300
				if object.propertyValue < 3
					PlayMusic(1)
				else
					if object.outOfBounds == 0
						PlayMusic(1)
					end if
				end if
				signpost.timeout = 0
				object.state = 3
				object[30].type = TypeName[Act Finish]
				object[30].drawOrder = 6
				object[30].priority = PRIORITY_ACTIVE
			end if
		end if
		break
	case 3
		if object[0].controlMode == -1
			object[0].right = 1
			if object[0].xvel < 0
				object[0].interaction = 1
			else
				object[0].interaction = 0
			end if
		end if
		temp0 = 40
		temp0 += screen.xcenter
		temp0 <<= 16
		temp0 += object.xpos
		if object[0].xpos > temp0
			object[0].xpos = temp0
			object[0].speed = 0
			object[0].xvel = 0
		end if
		if stage.player2Enabled == 1
			if object[1].controlMode == -1
				object[1].right = 1
			end if
			if object[1].xpos > temp0
				object[1].xpos = temp0
				object[1].speed = 0
				object[1].xvel = 0
			end if
		end if
		break
	case 4
		if stage.player2Enabled == 1
			temp0 = 40
			temp0 += screen.xcenter
			temp0 <<= 16
			temp0 += object.xpos
			object[1].right = 1
			object[1].controlMode = -1
			if object[1].xpos > temp0
				object[1].xpos = temp0
				object[1].speed = 0
				object[1].xvel = 0
			end if
		end if
		break
	end switch
end event
