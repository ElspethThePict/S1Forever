function processColor
	switch object.propertyValue
		case 0 //emulated
			//object.store &= 0xE0E0E0
			break
		case 1 //measured
			object.red = object.store
			object.red &= 0xE00000
			object.red >>= 3
			object.red *= 9
			
			object.blu = object.store
			object.blu &= 0x00E000
			object.blu >>= 3
			object.blu *= 9
			
			object.grn = object.store
			object.grn &= 0x0000E0
			object.grn >>= 3
			object.grn *= 9
			
			object.store = object.red
			object.store += object.blu
			object.store += object.grn
			break
		case 2 //actual			
			object.process = object.store
			object.process &= 0xE00000
			object.process >>= 21
			CallFunction(actualColor)
			object.red = object.process
			object.red <<= 16				
			
			object.process = object.store
			object.process &= 0x00E000
			object.process >>= 13
			CallFunction(actualColor)
			object.blu = object.process
			object.blu <<= 8
			
			object.process = object.store
			object.process &= 0xE0
			object.process >>= 5
			CallFunction(actualColor)
			object.grn = object.process				
			
			object.store = object.red
			object.store += object.blu
			object.store += object.grn
			break
	end switch
end function

function actualColor
	switch object.process

		case 5
			object.process = 172
			break
		case 6
			object.process = 206
			break
		case 7
			object.process = 255
			break
	end switch
end function
