event ObjectMain
	switch object.state
	case 0
		if object.value0 < 8
			object.value0++
		else
			object.value0 = 0
			object.state++
			object.frame = 1
		end if
		break
	case 1
		if object.value0 < 8
			object.value0++
		else
			object.value0 = 0
			object.state++
			object.frame = 2
		end if
		break
	case 2
		object.xpos += object.xvel
		object.ypos += object.yvel
		if object.value0 > 1
			object.frame = 3
		else
			object.frame = 2
		end if
		object.value0++
		object.value0 &= 3
		break
	end switch
	if object.outOfBounds == 1
		object.type = TypeName[Blank Object]
	end if
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		if object.state > 0
			BoxCollisionTest(C_TOUCH, object.entityPos, -6, -6, 6, 6, currentPlayer, object[currentPlayer].value38, object[currentPlayer].value40, object[currentPlayer].value39, object[currentPlayer].value41)
			if checkResult == 1
				CallFunction(PlayerObject_ProjectileHit)
			end if
		end if
	next
end event


event ObjectDraw
	DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
end event
