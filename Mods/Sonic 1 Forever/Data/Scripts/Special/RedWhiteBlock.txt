				if checkResult == 1
					object.propertyValue = 0
					object.inkEffect = INK_NONE
				end if

				BoxCollisionTest(C_TOUCH, object.entityPos, -160, -60, 160, -36, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
				if checkResult == 1
