	if object.state == 1
		object.frame = object.value0
		object.frame >>= 3
		object.value0++
		if object.value0 == 32
			object.value0 = 0
			object.state = 0
			object.frame = 0
		end if
	end if
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		CallFunction(SpecialSetup_Function10)
	next
end event

end event
