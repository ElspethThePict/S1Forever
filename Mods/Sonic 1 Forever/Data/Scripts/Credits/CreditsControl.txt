event ObjectMain
	switch object.state
	case 0
		object.value0 = 320
		SetScreenFade(0, 0, 0, object.value0)
		object.state++
		if credits.screen == 0
			PlayMusic(0)
		end if
		break
	case 1
		if object.value0 > 0
			object.value0 -= 8
		else
			object.alpha = 128
			object.state++
		end if
		SetScreenFade(0, 0, 0, object.value0)
		break
	case 2
		object.value0++
		if object.value0 == 136
			object.value0 = 0
			object.state++
		end if
		break
	case 3
		if object.value0 < 256
			object.value0 += 8
			SetScreenFade(0, 0, 0, object.value0)
		else
			object.state = 0
			credits.screen++
			credits.screen %= 10
			temp0 = credits.screen
			temp0 <<= 8
			screen.yoffset = temp0
			options.attractMode = 1
			switch credits.screen
			case 9
				object.state = 4
				if saved.superStates == 0
					if specialStage.emeralds >= 63
						screen.yoffset = 0xA00
						//stage.activeList = PRESENTATION_STAGE
						//stage.listPos = 5
						//LoadStage()
					else
						screen.yoffset = 0x900
					end if
				else
					if specialStage.emeralds >= 127
						screen.yoffset = 0xA00
						//stage.activeList = PRESENTATION_STAGE
						//stage.listPos = 5
						//LoadStage()
					else
						screen.yoffset = 0x900
					end if
				end if
				break
			case 1
			case 8
				stage.activeList = REGULAR_STAGE
				stage.listPos = 0
				LoadStage()
				break
			case 2
				stage.activeList = REGULAR_STAGE
				stage.listPos = 4
				LoadStage()
				break
			case 3
				stage.activeList = REGULAR_STAGE
				stage.listPos = 8
				LoadStage()
				break
			case 4
				stage.activeList = REGULAR_STAGE
				stage.listPos = 11
				LoadStage()
				break
			case 5
				stage.activeList = REGULAR_STAGE
				stage.listPos = 14
				LoadStage()
				break
			case 6
				stage.activeList = REGULAR_STAGE
				stage.listPos = 15
				LoadStage()
				break
			case 7
				stage.activeList = REGULAR_STAGE
				stage.listPos = 16
				LoadStage()
				break
			end switch
			SetScreenFade(0, 0, 0, 255)
		end if
		break
	case 4
		if object.value0 > 0
			object.value0 -= 8
		else
			object.alpha = 128
			object.state++
		end if
		SetScreenFade(0, 0, 0, object.value0)
		break
	case 5
		object.value0++
		if object.value0 == 0x4B0
			object.value0 = 0
			object.state++
		end if
		
		if inputPress.start == 1
			object.value0 = 0
			object.state = 6
		end if
		if inputPress.buttonA == 1
			object.value0 = 0
			object.state = 6
		end if
		if inputPress.buttonB == 1
			object.value0 = 0
			object.state = 6
		end if
		if inputPress.buttonC == 1
			object.value0 = 0
			object.state = 6
		end if
		break
	case 6
		if object.value0 < 256
			object.value0 += 8
			SetScreenFade(0, 0, 0, object.value0)
		else
			stage.activeList = PRESENTATION_STAGE
			stage.listPos = 0
			StopMusic()
			options.attractMode = 0
			SetScreenFade(0, 0, 0, 255)
			LoadStage()
		end if
		break
	end switch
	temp0 = screen.xcenter
	FlipSign(temp0)
	temp0 += 256
	screen.xoffset = temp0
	if saved.mobile == 1
		CheckTouchRect(0, 0, screen.xsize, screen.ysize)
		if checkResult > -1
			inputPress.start = 1
		end if
	end if
	if inputPress.start == 1
		if object.state < 6
			object.state = 6
		end if
	end if
end event
