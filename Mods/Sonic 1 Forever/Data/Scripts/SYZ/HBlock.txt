			break
		case 1

			break
		case 2

		case 3

			break
		end switch

event RSDKDraw
	DrawSprite(0)

	if editor.showGizmos == true
		editor.drawingOverlay = true
		
		// Draw the block's starting position

		switch object.propertyValue
		case 0
			object.value0 = 0x2000
			break
			
		case 1
			object.value0 = 0x1000
			break
			
		case 2
			object.value0 = -0x1000
			break
			
		case 3
			object.value0 = -0x2000
			break
			
		end switch

		object.inkEffect = 1
		Sin(object.value1, 0x80)
		object.value1 *= object.value0
		object.value1 += object.xpos
		object.value1 &= 0xFFFF0000

		DrawSpriteFX(0, FX_INK, object.value1, object.ypos)
		
		switch object.propertyValue
		case 0
		case 3
			// Draw an arrow for the block's full path
			FlipSign(object.value0)
			Sin(temp0, 0x80)
			temp0 *= object.value0
			temp0 += object.xpos
			temp0 &= 0xFFFF0000
			DrawArrow(object.value1, object.ypos, temp0, object.ypos, 255, 255, 255)
			break
			
		end switch
		
		editor.drawingOverlay = false
	end if
end event
