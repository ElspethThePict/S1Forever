event ObjectMain
	switch object.state
	case 0
		object.frame = object.animationTimer
		object.frame /= 3
		object.animationTimer++
		if object.animationTimer == 6
			object.animationTimer = 0
			object.value0++
			if object.value0 == 6
				object.value0 = 0
				object.state++
			end if
		end if
		break
	case 1
		object.frame = object.animationTimer
		object.frame /= 3
		object.frame += 2
		object.animationTimer++
		if object.animationTimer == 6
			object.animationTimer = 0
			object.value0++
			if object.value0 == 2
				object.type = TypeName[Blank Object]
			end if
		end if
		break
	end switch
end event


event ObjectDraw
	DrawSprite(object.frame)
end event
