function LavaGeyser_DebugSpawn
	CreateTempObject(TypeName[Lava Geyser], 0, object.xpos, object.ypos)
	object[tempObjectPos].value1 = object.ypos
	object[tempObjectPos].drawOrder = 5
end function


event ObjectMain
	switch object.state
	case 0
		foreach (TypeName[Push Block], arrayPos0, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, arrayPos0, -16, -16, 16, 16, object.entityPos, -16, -32, 16, 16)
			if checkResult == 1
				object.state = 1
				object.animationTimer = 0
				object.value2 = 0
				object.value4 = 4
			end if
		next
		break
	case 1
		object.value4 = object.value2
		object.value4 /= 3
		object.value4 += 4
		object.value2++
		object.value2 %= 6
		object.value0++
		if object.value0 == 12
			object.value0 = 0
			object.state++
		end if
		break
	case 2
		object.value4 = object.value2
		object.value4 /= 3
		object.value4 += 6
		object.value2++
		object.value2 %= 6
		object.value0++
		if object.value0 == 12
			object.value0 = 0
			object.frame = 0
			object.value3 = 8
			object.yvel = -0x54000
			object.state++
			PlaySfx(SfxName[Fire Burn], 0)
			foreach (TypeName[Push Block], arrayPos0, ACTIVE_ENTITIES)
				BoxCollisionTest(C_TOUCH, arrayPos0, -16, -16, 16, 16, object.entityPos, -32, -32, 32, 16)
				if checkResult == 1
					object[arrayPos0].yvel = -0x59000
					object[arrayPos0].gravity = GRAVITY_AIR
				end if
			next
		end if
		break
	case 3
		object.yvel += 0x1800
		object.ypos += object.yvel
		if object.ypos >= object.value1
			object.ypos = object.value1
			object.state++
		end if
		object.value4 = object.value2
		object.value4 /= 3
		object.value3 = object.value4
		object.value4 += 2
		object.value3 += 8
		object.value2++
		object.value2 %= 6
		object.frame = object.animationTimer
		object.frame >>= 3
		object.animationTimer++
		object.animationTimer &= 15
		temp0 = object.value1
		temp0 -= object.ypos
		temp0 >>= 16
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, object.entityPos, -32, 0, 32, temp0, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if checkResult == 1
				CallFunction(PlayerObject_FireHit)
			end if
		next
		break
	case 4
		object.value4 = object.value2
		object.value4 /= 3
		object.value4 += 2
		object.value2++
		object.value2 %= 6
		object.value0++
		if object.value0 == 12
			object.value0 = 0
			object.state++
		end if
		break
	case 5
		object.value4 = object.value2
		object.value4 /= 3
		object.value4 += 4
		object.value2++
		object.value2 %= 6
		object.value0++
		if object.value0 == 12
			object.value0 = 0
			object.state = 0
		end if
		break
	end switch
end event


event ObjectDraw
	switch object.state
	case 1
	case 2
	case 4
	case 5
		DrawSpriteXY(object.value4, object.xpos, object.value1)
		break
	case 3
		temp0 = object.value1
		temp0 -= object.ypos
		temp0 >>= 16
		temp1 = temp0
		FlipSign(temp1)
		if object.frame == 0
			EditFrame(0, -32, temp1, 64, temp0, 317, 1)
		else
			EditFrame(1, -32, temp1, 64, temp0, 382, 1)
		end if
		DrawSpriteXY(object.frame, object.xpos, object.value1)
		DrawSprite(object.value3)
		DrawSpriteXY(object.value4, object.xpos, object.value1)
		break
	end switch
end event
