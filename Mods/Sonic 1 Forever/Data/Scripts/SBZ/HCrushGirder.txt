event ObjectMain
	switch object.state
	case 0
		object.value0++
		if object.value0 == 96
			object.value0 = 6
			object.value3 = 1
			object.state = 4
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_BOX, object.entityPos, -96, -24, 96, 24, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			switch checkResult
			case 1
				object[currentPlayer].xpos += 0x10000
				break
			case 2
				object[currentPlayer].pushing = 0
				break
			end switch
		next
		object.xpos += 0x10000
		break
	case 1
		object.value0++
		if object.value0 == 48
			object.value0 = 6
			object.value3 = 2
			object.state = 4
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_BOX, object.entityPos, -96, -24, 96, 24, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			switch checkResult
			case 1
				object[currentPlayer].ypos += 0x10000
				break
			case 4
				if object[currentPlayer].gravity == GRAVITY_GROUND
					CallFunction(PlayerObject_Kill)
				end if
				break
			end switch
		next
		object.ypos += 0x10000
		break
	case 2
		object.value0++
		if object.value0 == 96
			object.value0 = 6
			object.value3 = 3
			object.state = 4
		end if
		object.ypos -= 0x4000
		object.xpos -= 0x10000
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_BOX, object.entityPos, -96, -24, 96, 24, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			switch checkResult
			case 1
				object[currentPlayer].xpos -= 0x10000
				break
			case 4
				break
			end switch
		next
		break
	case 3
		object.value0++
		if object.value0 == 24
			object.value0 = 6
			object.value3 = 0
			object.state = 4
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_BOX, object.entityPos, -96, -24, 96, 24, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			switch checkResult
			case 1
				object[currentPlayer].ypos -= 0x10000
				break
			case 4
				break
			end switch
		next
		object.ypos -= 0x10000
		break
	case 4
		if object.value0 > 0
			object.value0--
		else
			object.state = object.value3
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_BOX, object.entityPos, -96, -24, 96, 24, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if checkResult == 4
				if object[currentPlayer].gravity == GRAVITY_GROUND
					CallFunction(PlayerObject_Kill)
				end if
			end if
		next
	end switch
end event
