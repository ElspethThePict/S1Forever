private alias 67 : TYPE_RISINGPLATFORM

// Function declarations
reserve function RisingPlatform_DebugDraw
reserve function RisingPlatform_DebugSpawn

// Static Values

// Tables

function RisingPlatform_DebugDraw
	DrawSprite(0)
end function


function RisingPlatform_DebugSpawn
	CreateTempObject(TypeName[Rising Platform], 0, object.xpos, object.ypos)
	object[tempObjectPos].value3 = object.ypos
	object[tempObjectPos].drawOrder = 4
end function


event ObjectMain
	if object.value0 == 1
		if object.value2 == 0
			object.value2 = 30
		end if
		if object.value4 < 64
			object.value4 += 4
		end if
	else
		if object.value4 > 0
			object.value4 -= 4
		end if
	end if
	temp0 = object.ypos
	temp0 &= -0x10000
	switch object.state
	case 0
		temp1 = object.value4
		temp1 <<= 1
		Sin(temp2, temp1)
		temp2 <<= 9
		object.ypos = object.value3
		object.ypos += temp2
		if object.value2 != 0
			object.value2--
			if object.value2 == 0
				object.state++
				object.priority = PRIORITY_ACTIVE
			end if
		end if
		break
	case 1
		object.ypos += object.yvel
		object.yvel -= 0x800
		ObjectTileCollision(CSIDE_ROOF, 0, -12, 0)
		if checkResult == 1
			object.state++
		end if
		if object.outOfBounds == 1
			object.ypos = object.value3
			object.yvel = 0
			object.value2 = 0
			object.state = 3
		end if
		break
	case 2
		if object.outOfBounds == 1
			object.ypos = object.value3
			object.yvel = 0
			object.value2 = 0
			object.state++
		end if
		break
	case 3
		if object.outOfBounds == 1
			object.state = 0
			object.priority = PRIORITY_ACTIVE_BOUNDS
		end if
		break
	end switch
	object.value1 = object.ypos
	object.value1 &= -0x10000
	object.value1 -= temp0
	object.value0 = 0
	if object.state < 3
		temp1 = object.ypos
		object.ypos = temp0
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_BOX, object.entityPos, -32, -12, 32, 12, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if checkResult == 1
				object.value0 = 1
				object[currentPlayer].ypos += object.value1
			end if
		next
		object.ypos = temp1
	end if
end event


event ObjectDraw
	if object.state < 3
		DrawSprite(0)
	end if
end event


event ObjectStartup
	LoadSpriteSheet("LZ/Objects.gif")
	SpriteFrame(-32, -12, 64, 24, 126, 137)
	foreach (TypeName[Rising Platform], arrayPos0, ALL_ENTITIES)
		object[arrayPos0].value3 = object[arrayPos0].ypos
		object[arrayPos0].drawOrder = 4
	next
	SetTableValue(TypeName[Rising Platform], DebugMode_ObjCount, DebugMode_TypesTable)
	SetTableValue(RisingPlatform_DebugDraw, DebugMode_ObjCount, DebugMode_DrawTable)
	SetTableValue(RisingPlatform_DebugSpawn, DebugMode_ObjCount, DebugMode_SpawnTable)
	DebugMode_ObjCount++
end event

event RSDKDraw
	DrawSprite(0)
end event


event RSDKLoad
	LoadSpriteSheet("LZ/Objects.gif")
	SpriteFrame(-32, -12, 64, 24, 126, 137)
	
	SetVariableAlias(ALIAS_VAR_PROPVAL, "unused")
end event
