event ObjectMain
	if object[0].yvel >= 0
		if screen.cameraY >= 0x878
			temp1 = object[0].ypos
			foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
				temp0 = object[currentPlayer].ypos
				temp0 -= temp1
				Abs(temp0)
				if temp0 < 0x8C0000
					object[currentPlayer].ypos -= 0x8000000
				end if
			next
			screen.cameraY -= 0x800
			screen.yoffset -= 0x800
		end if
	end if
	if object[0].yvel <= 0
		if screen.cameraY <= 120
			temp1 = object[0].ypos
			foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
				temp0 = object[currentPlayer].ypos
				temp0 -= temp1
				Abs(temp0)
				if temp0 < 0x8C0000
					object[currentPlayer].ypos += 0x8000000
				end if
			next
			screen.cameraY += 0x800
			screen.yoffset += 0x800
		end if
	end if
end event

	DrawSprite(0)
