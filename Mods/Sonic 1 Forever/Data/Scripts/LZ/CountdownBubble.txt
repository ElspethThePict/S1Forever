event ObjectMain
	switch object.state
	case 0
		object.ypos += object.yvel
		object.value0++
		if object.value0 > 6
			object.value0 = 0
			object.frame++
		end if
		if object.frame < 5
			Sin(object.xpos, object.angle)
			object.xpos <<= 9
			object.xpos += object.value1
			object.angle += 4
			object.angle &= 511
		else
			temp0 = object.propertyValue
			temp0 *= 3
			object.frame += temp0
			object.value1 = object.ixpos
			object.value1 -= screen.xoffset
			object.value2 = object.iypos
			object.value2 -= screen.yoffset
			object.state++
		end if
		break
	case 1
		object.value0++
		if object.value0 == 6
			object.value0 = 0
			object.frame++
			object.state++
		end if
		break
	case 2
		object.value0++
		if object.value0 == 6
			object.value0 = 0
			object.frame++
			object.state++
		end if
		break
	case 3
		object.value0++
		if object.value0 == 8
			object.value0 = 0
			object.frame--
			object.state++
		end if
		break
	case 4
		object.value0++
		if object.value0 == 8
			object.value0 = 0
			object.frame++
			object.state++
		end if
		break
	case 5
		object.value0++
		if object.value0 == 8
			object.value0 = 0
			object.frame--
			object.state++
		end if
		break
	case 6
		object.value0++
		if object.value0 == 8
			object.value0 = 0
			object.frame++
			object.state++
		end if
		break
	case 7
		object.value0++
		if object.value0 == 8
			object.value0 = 0
			object.frame--
			object.state++
		end if
		break
	case 8
		object.value0++
		if object.value0 == 9
			object.type = TypeName[Blank Object]
		end if
		break
	end switch
end event


event ObjectDraw
	if object.state == 0
		DrawSprite(object.frame)
	else
		DrawSpriteScreenXY(object.frame, object.value1, object.value2)
	end if
end event
