event ObjectMain
	if object.value0 != LZSetup_value30
		object.direction ^= FLIP_X
		object.value0 = LZSetup_value30
		temp2 = 1
	else
		object.frame = oscillation
		object.frame >>= 2
		object.frame &= 3
		if object.direction == FLIP_X
			object.frame ^= 3
		end if
	end if
end event

	DrawSprite(object.frame)
end event
