private alias 47 : TYPE_POCKY

// Function declarations
reserve function Pocky_DebugDraw
reserve function Pocky_DebugSpawn

// Static Values

// Tables

function Pocky_DebugDraw
	DrawSprite(2)
end function


function Pocky_DebugSpawn
	CreateTempObject(TypeName[Pocky], 0, object.xpos, object.ypos)
	GetBit(temp0, object.direction, 0)
	if temp0 == 0
		object[tempObjectPos].state = 3
		object[tempObjectPos].xvel = 0x20000
		object[tempObjectPos].yvel = -0x40000
		Rand(checkResult, 32)
		if checkResult < 16
			object[tempObjectPos].direction = FLIP_X
			FlipSign(object[tempObjectPos].xvel)
		end if
	else
		object[tempObjectPos].state = 4
	end if
end function


event ObjectMain
	switch object.state
	case 0
		if object.value0 < object.propertyValue
			object.value0++
		else
			object.xvel = 0x20000
			if object.value1 == 1
				object.drawOrder = 4
				Rand(checkResult, 32)
				if checkResult < 16
					object.direction = FLIP_X
					FlipSign(object.xvel)
				end if
			else
				object.direction = FLIP_X
				FlipSign(object.xvel)
			end if
			object.yvel = -0x40000
			object.state++
		end if
		break
	case 1
		object.ypos += object.yvel
		object.yvel += 0x3800
		if object.yvel > 0
			ObjectTileCollision(CSIDE_FLOOR, 0, 12, 0)
			if checkResult == 1
				object.yvel = -0x40000
				object.state++
			end if
		end if
		break
	case 4
		if object[0].xpos < object.xpos
			object.direction = FLIP_X
		else
			object.direction = FLIP_NONE
		end if
	case 2
	case 3
	case 5
		object.xpos += object.xvel
		object.ypos += object.yvel
		object.yvel += 0x3800
		if object.yvel > 0
			ObjectTileCollision(CSIDE_FLOOR, 0, 12, 0)
			if checkResult == 1
				if object.state == 3
					if object.value0 == 1
						FlipSign(object.xvel)
						object.direction ^= FLIP_X
					end if
					object.value0 ^= 1
				end if
				object.yvel = -0x40000
			end if
		end if
		break
	end switch
	if object.state < 3
		if object.outOfBounds == 1
			object.type = TypeName[Blank Object]
		end if
	end if
end event


event ObjectDraw
	switch object.state
	case 0
	case 1
		DrawSprite(2)
		break
	case 2
	case 3
	case 4
	case 5
		if object.yvel < 0
			DrawSpriteFX(0, FX_FLIP, object.xpos, object.ypos)
		else
			DrawSpriteFX(1, FX_FLIP, object.xpos, object.ypos)
		end if
		break
	end switch
end event


event ObjectStartup
	LoadSpriteSheet("Global/Items.gif")
	SpriteFrame(-8, -12, 16, 24, 183, 26)
	SpriteFrame(-8, -12, 16, 24, 183, 51)
	SpriteFrame(-7, -12, 16, 24, 183, 1)
	foreach (TypeName[Pocky], arrayPos0, ALL_ENTITIES)
		if object[arrayPos0].propertyValue == 1
			object[arrayPos0].state = 3
			object[arrayPos0].xvel = 0x20000
			object[arrayPos0].yvel = -0x40000
			Rand(checkResult, 32)
			if checkResult < 16
				object[arrayPos0].direction = FLIP_X
				FlipSign(object[arrayPos0].xvel)
			end if
		else
			if object[arrayPos0].propertyValue == 2
				object[arrayPos0].state = 4
			end if
			if object[arrayPos0].propertyValue == 3
				object[arrayPos0].state = 5
				object[arrayPos0].xvel = 0x20000
				FlipSign(object[arrayPos0].xvel)
				object[arrayPos0].direction = FLIP_X
			end if
		end if
		object[arrayPos0].propertyValue = 0
	next
	SetTableValue(TypeName[Pocky], DebugMode_ObjCount, DebugMode_TypesTable)
	SetTableValue(Pocky_DebugDraw, DebugMode_ObjCount, DebugMode_DrawTable)
	SetTableValue(Pocky_DebugSpawn, DebugMode_ObjCount, DebugMode_SpawnTable)
	DebugMode_ObjCount++
end event

event RSDKEdit
	if editor.returnVariable == true
		switch editor.variableID
		case EDIT_VAR_PROPVAL // property value
			checkResult = object.propertyValue
			break
			
		case 0 // behaviour
			checkResult = object.propertyValue
			break
			
		end switch
	else
		switch editor.variableID
		case EDIT_VAR_PROPVAL // property value
			object.propertyValue = editor.variableValue
			break
			
		case 0 // behaviour
			object.propertyValue = editor.variableValue
			break
			
		end switch
	end if
end event


event RSDKDraw
	DrawSprite(0)
end event

	SetActiveVariable("behavior")

end event
