function modes.offset
	modes.ymin = modes.highlight
	modes.ymin *= -73
	modes.ymin += 32
	modes.ymax = modes.ymin
	modes.ymax += 135
	
	if modes.yoffset > modes.ymax
		modes.yoffset -= 3
		if modes.yoffset < modes.ymax
			modes.yoffset = modes.ymax
		end if
	else
		if modes.yoffset < modes.ymin
			modes.yoffset += 3
			if modes.yoffset > modes.ymin
				modes.yoffset = modes.ymin
			end if
		end if
	end if
end function

function modes.control
	if inputPress.up == 1
		CallFunction(main.sound)
		modes.highlight--
		while modes.highlight < 0
			modes.highlight += modes.count
		loop
	else
		if inputPress.down == 1
			CallFunction(main.sound)
			modes.highlight++
			modes.highlight %= modes.count
		else
			if inputPress.buttonB == 1
				object.state = modes.exit
				menu.state = 1
			else
				CallFunction(main.soundconf)
				temp0 = inputPress.buttonA
				temp0 |= inputPress.start
				if temp0 == 1
					object.state = modes.load
				end if
			end if
		end if
	end if
end function

function modes.exit
	modes.extime++
	if modes.extime >= 30
		ResetObjectEntity(object.entityPos, TypeName[Blank Object], 0, 0, 0)
	end if
end function

function modes.load
	modes.extime += 16
	music.volume -= 4
	SetScreenFade(0, 0, 0, modes.extime)
	if modes.extime >= 384
		StopMusic()
		switch modes.highlight
			case 0 //Time Attack
				stage.activeList = PRESENTATION_STAGE
				stage.listPos = 10
				LoadStage()
				break
			case 1 //Arcade
				stage.activeList = PRESENTATION_STAGE
				stage.listPos = 6
				LoadStage()
				break
			case 2 //Golf
				stage.activeList = PRESENTATION_STAGE
				stage.listPos = 11
				LoadStage()
				break
		end switch
	end if
end function

event ObjectMain
	if object.state == 0
		object.state = modes.control
		modes.yoffset = 32
	else
		CallFunction(object.state)	
	end if
	CallFunction(modes.offset)
end event

						end if

				GetTextInfo(temp7, 1, TEXTINFO_TEXTDATA, modes.drawline, temp3)
