function display.menu
	object.curMenu = 0
	switch engine.platformID
	case RETRO_WIN:
	case RETRO_OSX:
	case RETRO_XBOX_360:
	case RETRO_PS3:
		CallFunction(displaystndrd.menu)
		break
	case RETRO_iOS:
	case RETRO_ANDROID:
	case RETRO_WP7:
		CallFunction(displaymob.menu)
		break
	case RETRO_UWP:
	case RETRO_LINUX:
		CallFunction(displaystndrd.menu)
		break
	case RETRO_SWITCH:
		CallFunction(displaymob.menu)
		break
	end switch
end function

function displaystndrd.menu
	CallFunction(main.sound)
		if inputPress.up == 1
			menu.highlight--
			if menu.highlight < 0
				if saved.mobile == 1
					menu.highlight = 9
				else
					menu.highlight = 8
				end if
			end if
		else	
			if inputPress.down == 1
				menu.highlight++
				if saved.mobile == 1
					if menu.highlight > 9
						menu.highlight = 0
					end if
				else
					if menu.highlight > 8
						menu.highlight = 0
					end if
				end if
			end if
		end if
		
		if inputPress.buttonL == 1
			object.state = datamang.menu
			menu.highlight = 0
		end if
		if inputPress.buttonR == 1
			object.state = gameplay.menu
			menu.highlight = 0
		end if
		

		switch menu.highlight
			case 0 //Display Highlighted
				if inputPress.left == 1
					object.state = datamang.menu
				end if
				if inputPress.right == 1
					object.state = gameplay.menu
				end if
				break
			case 1 // Fullscreen
				temp0 = inputPress.left
				temp0 |= inputPress.right
				if temp0 == 1
					saved.fullscreen ^= 1
					CallNativeFunction2(SetWindowFullScreen, saved.fullscreen, 0)
				end if
				break
			case 2 // Screen Width		
				if inputPress.left == 1
					switch screen.xsize
						case 320
							CallNativeFunction2(SetScreenWidth, 384, 0)
							break
						case 384
							CallNativeFunction2(SetScreenWidth, 426, 0)
							break
						case 426
							CallNativeFunction2(SetScreenWidth, 320, 0)
							break
						default
							CallNativeFunction2(SetScreenWidth, 426, 0)
							break
					end switch
					CallFunction(sidebar.adj)
				end if
				if inputPress.right == 1
					switch screen.xsize
						case 320
							CallNativeFunction2(SetScreenWidth, 426, 0)
							break
						case 384
							CallNativeFunction2(SetScreenWidth, 320, 0)
							break
						case 426
							CallNativeFunction2(SetScreenWidth, 384, 0)
							break
						default
							CallNativeFunction2(SetScreenWidth, 426, 0)
							break
					end switch
					CallFunction(sidebar.adj)
				end if
				break
			case 3 // Window Scale		
				if inputPress.left == 1					
					saved.scale--
					if saved.scale < 1
						saved.scale = 4
					end if
					CallNativeFunction2(SetWindowScale, saved.scale, 0)
				end if
				if inputPress.right == 1
					saved.scale++
					if saved.scale > 4
						saved.scale = 1
					end if
					CallNativeFunction2(SetWindowScale, saved.scale, 0)
				end if
				break
			case 4 // Graphics Mode
				temp0 = 0
				if inputPress.left == 1
					temp0 = 1
					saved.graphics--
					if saved.graphics < 0
						saved.graphics = 2
					end if
				end if
				if inputPress.right == 1
					temp0 = 1
					saved.graphics++
					if saved.graphics > 2
						saved.graphics = 0
					end if
				end if
				if temp0 == 1
					LoadPalette("MainMenu.act", 0, 0, 0, 256)
					CallFunction(emerald_color)
					CreateTempObject(TypeName[Palette Police], saved.graphics, 0, 0)
				end if				
				break
			case 5 // Language
				if inputPress.left == 1
					saved.language--
					if saved.language < 0
						saved.language = 3
					end if
					CallFunction(Lang_Init)
				end if
				if inputPress.right == 1
					saved.language++
					if saved.language > 3
						saved.language = 0
					end if
					CallFunction(Lang_Init)
				end if				
				break
			case 6 // VSync
				temp0 = inputPress.left
				temp0 |= inputPress.right
				if temp0 == 1
					saved.vsync ^= 1
					CallNativeFunction2(SetWindowVSync, saved.vsync, 0)
				end if
				break
			case 7 // Scaling Mode
				if inputPress.left == 1					
					saved.scalemode--
					if saved.scalemode < 0
						saved.scalemode = 3
					end if
					CallNativeFunction2(SetWindowScaleMode, saved.scalemode, 0)
				end if
				if inputPress.right == 1
					saved.scalemode++
					if saved.scalemode > 3
						saved.scalemode = 0
					end if
					CallNativeFunction2(SetWindowScaleMode, saved.scalemode, 0)
				end if
				break
			case 8 // Refresh Rate
				temp0 = inputPress.left
				temp0 |= inputPress.right
				if temp0 == 1
					if saved.fps == 60
						saved.fps = 50
					else
						saved.fps = 60
					end if
					CallNativeFunction2(SetFrameRate, saved.fps, 0)
				end if
				break
			case 9 // Button Visability 
				if inputPress.left == 1	
					touch.fade = 0				
					saved.transmobile--
					if saved.transmobile < 0
						saved.transmobile = 4
					end if
				end if
				if inputPress.right == 1
					touch.fade = 0	
					saved.transmobile++
					if saved.transmobile > 4
						saved.transmobile = 0
					end if
				end if
				break
		end switch
end function

	end if

		object.prevState = object.state

			object.drawState = display.draw
		else

				object.drawState = gameplay.draw

						else
							if object.state == datamang.menu
								object.drawState = datamang.draw
							end if
					end if
				end if
			end if
		end if
	end if
end event
