event ObjectMain

	switch object.state
	case 0
		stage[0].activeLayer = 2
		object.state++
		break
	case 1
		object.value0++
		if object.value0 > 7
			tileLayer[2].deformationOffset++
			object.value0 = 0
		end if
		break
	end switch
	hParallax[1].scrollSpeed = 0
	hParallax[2].scrollSpeed = 0
	if screen.xsize == 320
		temp0 = 172
	else
		temp0 = 119
	end if
	temp0 <<= 16
	hParallax[1].scrollPos = temp0
	hParallax[2].scrollPos = temp0
	
	temp0 = screen.xcenter
	FlipSign(temp0)
	temp0 += 256
	screen.xoffset = temp0
	
	object.value2++
	if object.value2 == 420
		PlayMusic(1)
	end if
	if object.value2 >= 620
		object.value1++
		object.value2 = 590
	end if
	
	if object.value1 >= 20
		if object.value3 < 320
			object.value3 += 8
			music.volume -= 2
		else
			StopMusic()
			stage.activeList = PRESENTATION_STAGE
			stage.listPos = StageName[P - MAIN MENU]
			LoadStage()
		end if
		SetScreenFade(0, 0, 0, object.value3)
	end if

			CheckEqual(inputPress.start, 1)
			temp0 = checkResult
			CheckEqual(inputPress.buttonA, 1)
			temp0 |= checkResult
			CheckEqual(inputPress.buttonB, 1)
			temp0 |= checkResult
			CheckEqual(inputPress.buttonC, 1)
			temp0 |= checkResult
			if temp0 == 1
			stage.activeList = PRESENTATION_STAGE
			stage.listPos = StageName[P - MAIN MENU]
			LoadStage()
			end if
	
end event
